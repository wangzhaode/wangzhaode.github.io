自定义QQ5.0侧滑菜单改进版

关于QQ5.0的侧滑估计大家现在在网上能看到很多教程与实现方法，我发现有一篇写的十分简单易懂，而且在慕课网上还有对应视频，感觉着一个教程应该是最适合大家学习的啦，在此十分感谢原创作者。不过其中也有一点待改进的地方，那就是在滑动式的判断上，他只对滑动距离做了判断，但是未对滑动速度做判断，导致使用滑动菜单式感觉十分生硬，所以我在此处添加了速度判断，希望能让这个教程更加完美。

想细致学习的同学可以去博主那里看一下博客，或者是去慕课网看一下视频，都是很不错的，首先了解了基本的滑动菜单实现，然后再来看如何添加速度判断就比较容易理解了。

应用到自定义的SlidingMenu类中

1.首先我们定义这几个变量

//用于求滑动速度的实例 id为点击触点ID
private VelocityTracker mVelocityTracker;
private int mPointerId;
//0表示未达到速度阈值；1表示达到向右滑动的阈值；2表示达到向左滑动的阈值
private int slidstate = 0;

2.给原来的onTouchEvent方法增加速度判断

经过不断尝试，大概把 2~3像素/秒 设置为阈值可以达到理想的自然滑动效果，所以在改变slidedata的这两个判断中使用了3.0作为阈值，

if (velocityX >= 3.0&&slidstate==0) {

slidstate = 2;

}

if (velocityX <= -3.0&&slidstate==0) {

slidstate = 1;

}

以下为改进后的onTouchEvent方法

[sourcecode language=”java”]

public boolean onTouchEvent(MotionEvent ev) {

//首先获得VelocityTracker类实例

if (mVelocityTracker == null) {

mVelocityTracker = VelocityTracker.obtain();//获得VelocityTracker类实例

}

int action = ev.getAction();

mVelocityTracker.addMovement(ev);//将事件加入到VelocityTracker类实例中

final VelocityTracker verTracker = mVelocityTracker;

switch (action) {

case MotionEvent.ACTION_DOWN:

mPointerId = ev.getPointerId(0);

break;

case MotionEvent.ACTION_MOVE:

//求伪瞬时速度

verTracker.computeCurrentVelocity(1,1000);

final float velocityX = verTracker.getXVelocity(mPointerId);

Log.d("sudu", velocityX+"");

//改变slidstate的状态

if (velocityX &gt;= 3.0&amp;&amp;slidstate==0) {

slidstate = 2;

}

if (velocityX &lt;= -3.0&amp;&amp;slidstate==0) {

slidstate = 1;

}

break;

case MotionEvent.ACTION_UP:

releaseVelocityTracker();

int scrollX = getScrollX();

Log.d("slidstate", slidstate+"");

switch (slidstate) {

case 0:

if (scrollX &gt;= mScreenWidth / 2) {

this.smoothScrollTo(mMenuWidth, 0);

isOpen = false;

} else {

this.smoothScrollTo(0, 0);

isOpen = true;

}

break;

case 1:

this.smoothScrollTo(mMenuWidth, 0);

isOpen = false;

break;

case 2:

this.smoothScrollTo(0, 0);

isOpen = true;

break;

}

slidstate=0;

return true;

}

return super.onTouchEvent(ev);

}

//其中释放releaseVelocityTracker的方法为

private void releaseVelocityTracker() {
if(null != mVelocityTracker) {
mVelocityTracker.clear();
mVelocityTracker.recycle();
mVelocityTracker = null;
}
}
